## Code
#include <stdio.h>  
int main()  
{  
   int num;  
   printf("I like egg tarts.");  
   printf("How many egg tarts have I eaten today?\n);  
   scanf("%d", &num);  
   printf("OMG! There's totally %d kcal !!!\n", num*217);  
   return 0;  
}
